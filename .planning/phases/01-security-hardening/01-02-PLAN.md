**Step 2 — Update app.ts:**

Replace the current `app.ts` with this implementation:

```typescript
import express, { type Express } from "express";
import cors from "cors";
import session from "express-session";
import { doubleCsrf } from "csrf-csrf";
import pinoHttp from "pino-http";
import router from "./routes";
import { logger } from "./lib/logger";

const app: Express = express();

// --- Logging ---
app.use(pinoHttp({ /* existing serializer config */ }));

// --- CORS: restrict to known origins ---
const allowedOrigins = process.env.ALLOWED_ORIGINS
  ? process.env.ALLOWED_ORIGINS.split(",").map((o) => o.trim())
  : ["http://localhost:5173"];

app.use(cors({
  origin: allowedOrigins,
  credentials: true,  // required for session cookies
}));

// --- Body parsing ---
app.use(express.json({ limit: "10mb" }));
app.use(express.urlencoded({ extended: true, limit: "10mb" }));

// --- Session ---
const SESSION_SECRET = process.env.SESSION_SECRET;
if (!SESSION_SECRET && process.env.NODE_ENV === "production") {
  throw new Error("SESSION_SECRET env var is required in production");
}

app.use(session({
  secret: SESSION_SECRET ?? "dev-secret-change-in-production",
  resave: false,
  saveUninitialized: false,
  cookie: {
    httpOnly: true,
    sameSite: "strict",
    secure: process.env.NODE_ENV === "production",  // HTTPS only in prod
    maxAge: 7 * 24 * 60 * 60 * 1000,               // 7 days
  },
}));

// --- CSRF (double-submit cookie, protects mutations) ---
const { generateToken, doubleCsrfProtection } = doubleCsrf({
  getSecret: () => SESSION_SECRET ?? "dev-secret-change-in-production",
  cookieName: "x-csrf-token",
  cookieOptions: {
    httpOnly: false,   // Frontend JS must be able to read this cookie to send in header
    sameSite: "strict",
    secure: process.env.NODE_ENV === "production",
  },
  size: 64,
  getTokenFromRequest: (req) =>
    (req.headers["x-csrf-token"] as string) ?? req.body?._csrf,
});

// Expose generateToken on app.locals so auth routes can issue it
app.locals.generateCsrfToken = generateToken;

// Apply CSRF protection to all state-changing methods
// Exclude: GET, HEAD, OPTIONS (safe methods), and /auth/login + /auth/register
// (login/register are CSRF-safe because they don't operate on session state the attacker has)
app.use((req, res, next) => {
  const safeMethods = ["GET", "HEAD", "OPTIONS"];
  const csrfExempt = ["/api/auth/login", "/api/auth/register", "/api/public/"];
  if (
    safeMethods.includes(req.method) ||
    csrfExempt.some((path) => req.path.startsWith(path))
  ) {
    return next();
  }
  return doubleCsrfProtection(req, res, next);
});

app.use("/api", router);

export default app;
```

```typescript
router.post("/auth/login", async (req, res): Promise<void> => {
  // ... existing email/password validation ...
  const passwordMatch = await bcrypt.compare(password, user.passwordHash);
  if (!passwordMatch) {
    res.status(401).json({ error: "INVALID_CREDENTIALS" });
    return;
  }

  // Set session
  req.session.userId = user.id;

  // Issue CSRF token for subsequent mutations
  const csrfToken = (req.app.locals.generateCsrfToken as (req: any, res: any) => string)(req, res);

  res.json({
    id: user.id,
    email: user.email,
    name: user.name,
    avatarUrl: user.avatarUrl ?? null,
    createdAt: user.createdAt.toISOString(),
    csrfToken,  // Frontend stores this and sends as X-CSRF-Token header
  });
});
```

```typescript
router.get("/auth/me", async (req, res): Promise<void> => {
  const userId = req.session?.userId;
  if (!userId) {
    res.status(401).json({ error: "Unauthorized" });
    return;
  }
  const [user] = await db.select().from(usersTable).where(eq(usersTable.id, userId));
  if (!user) {
    res.status(401).json({ error: "Unauthorized" });
    return;
  }
  // Issue a fresh CSRF token on each /me call (lets frontend refresh token after page reload)
  const csrfToken = (req.app.locals.generateCsrfToken as (req: any, res: any) => string)(req, res);
  res.json({
    id: user.id,
    email: user.email,
    name: user.name,
    avatarUrl: user.avatarUrl ?? null,
    createdAt: user.createdAt.toISOString(),
    csrfToken,
  });
});
```

3. **POST /auth/logout** — destroy session:
```typescript
router.post("/auth/logout", (req, res): void => {
  req.session.destroy((err) => {
    if (err) {
      res.status(500).json({ error: "Failed to logout" });
      return;
    }
    res.clearCookie("connect.sid");
    res.json({ ok: true });
  });
});
```

**Part B — Update routes/index.ts to add requireAuth middleware:**

Add a `requireAuth` middleware function and apply it to all `/organizations/*` routes. The key constraint: public routes (`/public/`, `/auth/`, `/health`, `/healthz`) must remain open.

```typescript
import { type Request, type Response, type NextFunction } from "express";

// Auth guard: applied to all /organizations/* routes
function requireAuth(req: Request, res: Response, next: NextFunction): void {
  if (!req.session?.userId) {
    res.status(401).json({ error: "Unauthorized" });
    return;
  }
  next();
}

// Apply to all organization-scoped routes ONLY
// (auth, health, public routes remain open)
router.use("/organizations", requireAuth);
```

Add this BEFORE the existing `router.use(organizationsRouter)` and other org-scoped routers. The path prefix `/organizations` matches `/organizations/*` so all routes under it (events, guests, campaigns, etc.) are protected.

Note: `teamRouter`, `dashboardRouter`, `remindersRouter`, `integrationsRouter`, `sendingDomainsRouter`, `plansRouter`, `emailProviderRouter` may also have `/organizations/:orgId` paths — check their route prefixes and apply `requireAuth` as a middleware on the router as well, or scope them all explicitly. The safest approach: apply requireAuth globally EXCEPT for the exempt paths:

```typescript
// Block everything except auth + health + public
router.use((req, res, next) => {
  const openPaths = ["/auth/", "/healthz", "/health", "/public/"];
  if (openPaths.some((p) => req.path.startsWith(p))) return next();
  return requireAuth(req, res, next);
});
```

Place this BEFORE all the `router.use(someRouter)` calls.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → POST /auth/login | Credentials cross here; session established |
| Browser (with session cookie) → /api/organizations/* | Authenticated requests; session validated per request |
| External origin → any mutation endpoint | Cross-site request; CSRF token required |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Elevation of Privilege | All /organizations/* routes unauthenticated | mitigate | requireAuth middleware checks req.session.userId, returns 401 if absent (per SEC-01) |
| T-02-02 | Spoofing | GET /auth/me hardcoded userId=1 | mitigate | Read userId from express-session, not hardcoded constant |
| T-02-03 | Cross-Site Request Forgery | Cookie-based session allows CSRF | mitigate | csrf-csrf double-submit token required for all mutations except login/register/public (per SEC-03) |
| T-02-04 | Information Disclosure | Open CORS allows any origin to read API | mitigate | CORS restricted to ALLOWED_ORIGINS env var; defaults to localhost:5173 |
| T-02-05 | Denial of Service | Session secret missing in production | mitigate | Throw at startup if SESSION_SECRET unset and NODE_ENV=production |
| T-02-06 | Elevation of Privilege | Session fixation | accept | express-session regenerates session ID on login (default behavior); low risk for v1 solo-developer timeline |
