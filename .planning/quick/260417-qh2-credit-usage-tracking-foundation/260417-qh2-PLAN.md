<objective>
Add the credit balance + usage-tracking foundation so HypeSpace can meter AI calls per organization. This is the monetization backbone for v2: every AI text/image call checks a balance, decrements on success, and writes an auditable usage_events row. Non-AI flows are untouched.

Purpose: Without this, we cannot charge by usage — every AI call today is effectively free and unbounded. With this, plan tiers enforce real economic limits (free=50, starter=500, growth=2500, agency=10000 credits/mo) and we have per-call audit trail for invoicing and debugging.

Output:
- 2 new organization columns + 1 new `usage_events` table (via drizzle push)
- `lib/credits.ts` with gate + debit + refund + summary helpers
- Both `/ai-generate` endpoints gated by credits (HTTP 402 when insufficient)
- `GET /organizations/:orgId/usage` extended to return credits summary
- OpenAPI spec updated so client gets generated types
- Backfill migration so existing seed org (id=1) starts with credits matching its plan
</objective>

```typescript
import { db, organizationsTable, usageEventsTable } from "@workspace/db";
import { eq, and, gte, sql, desc } from "drizzle-orm";

export type UsageKind = "ai_text" | "ai_image" | "ai_video" | "ai_transcription";

const COSTS: Record<UsageKind, number> = {
  ai_text: 2,
  ai_image: 10,
  ai_video: 100,         // reserved
  ai_transcription: 1,   // per minute — unused today
};

export function creditCost(kind: UsageKind): number {
  return COSTS[kind];
}

export async function hasCredits(
  orgId: number,
  kind: UsageKind,
): Promise<{ allowed: boolean; balance: number; required: number }> {
  const required = creditCost(kind);
  const [org] = await db
    .select({ balance: organizationsTable.creditsBalance })
    .from(organizationsTable)
    .where(eq(organizationsTable.id, orgId));
  const balance = org?.balance ?? 0;
  return { allowed: balance >= required, balance, required };
}

export interface DebitArgs {
  orgId: number;
  userId?: number | null;
  kind: UsageKind;
  provider: string;
  model?: string | null;
  costUsd?: number | null;
  metadata?: Record<string, unknown>;
}

export async function debitCredits(args: DebitArgs): Promise<void> {
  const credits = creditCost(args.kind);
  await db.transaction(async (tx) => {
    await tx
      .update(organizationsTable)
      .set({ creditsBalance: sql`${organizationsTable.creditsBalance} - ${credits}` })
      .where(eq(organizationsTable.id, args.orgId));
    await tx.insert(usageEventsTable).values({
      organizationId: args.orgId,
      userId: args.userId ?? null,
      kind: args.kind,
      credits,                          // positive = debit
      provider: args.provider,
      model: args.model ?? null,
      costUsd: args.costUsd != null ? args.costUsd.toString() : null,
      metadata: args.metadata ?? {},
    });
  });
}

export async function refundCredits(
  orgId: number,
  kind: UsageKind,
  reason: string,
): Promise<void> {
  const credits = creditCost(kind);
  await db.transaction(async (tx) => {
    await tx
      .update(organizationsTable)
      .set({ creditsBalance: sql`${organizationsTable.creditsBalance} + ${credits}` })
      .where(eq(organizationsTable.id, orgId));
    await tx.insert(usageEventsTable).values({
      organizationId: orgId,
      userId: null,
      kind,
      credits: -credits,                // negative = refund
      provider: "system",
      model: null,
      costUsd: null,
      metadata: { refundReason: reason },
    });
  });
}

export async function getOrgUsage(
  orgId: number,
  opts: { since?: Date } = {},
): Promise<{
  balance: number;
  limit: number;
  periodStart: string;
  used: number;
  byKind: Record<string, number>;
  recentEvents: Array<{
    id: number;
    kind: string;
    credits: number;
    provider: string;
    model: string | null;
    createdAt: string;
  }>;
}> {
  const [org] = await db
    .select({
      balance: organizationsTable.creditsBalance,
      limit: organizationsTable.creditsLimit,
      periodStart: organizationsTable.creditsPeriodStart,
    })
    .from(organizationsTable)
    .where(eq(organizationsTable.id, orgId));

  if (!org) {
    return {
      balance: 0, limit: 0, periodStart: new Date(0).toISOString(),
      used: 0, byKind: {}, recentEvents: [],
    };
  }

  const since = opts.since ?? org.periodStart;
  const rows = await db
    .select()
    .from(usageEventsTable)
    .where(and(
      eq(usageEventsTable.organizationId, orgId),
      gte(usageEventsTable.createdAt, since),
    ))
    .orderBy(desc(usageEventsTable.createdAt));

  const byKind: Record<string, number> = {};
  let used = 0;
  for (const r of rows) {
    byKind[r.kind] = (byKind[r.kind] ?? 0) + r.credits;
    used += r.credits;
  }

  return {
    balance: org.balance,
    limit: org.limit,
    periodStart: org.periodStart.toISOString(),
    used,
    byKind,
    recentEvents: rows.slice(0, 25).map((r) => ({
      id: r.id,
      kind: r.kind,
      credits: r.credits,
      provider: r.provider,
      model: r.model,
      createdAt: r.createdAt.toISOString(),
    })),
  };
}
```
